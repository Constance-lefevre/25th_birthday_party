    const buttons = document.querySelectorAll('#lang-switcher button');
    buttons.forEach(button => {
      button.addEventListener('click', () => {
        const lang = button.getAttribute('data-lang');
        buttons.forEach(btn => btn.classList.remove('active'));
        button.classList.add('active');

        if(lang === 'fr') {
          document.querySelectorAll('.lang-fr').forEach(el => el.style.display = '');
          document.querySelectorAll('.lang-eng').forEach(el => el.style.display = 'none');
        } else {
          document.querySelectorAll('.lang-fr').forEach(el => el.style.display = 'none');
          document.querySelectorAll('.lang-eng').forEach(el => el.style.display = '');
        }
      });
    });
